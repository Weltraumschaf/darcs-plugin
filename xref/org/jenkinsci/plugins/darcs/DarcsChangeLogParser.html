10  package org.jenkinsci.plugins.darcs;
11  
12  import hudson.model.AbstractBuild;
13  import hudson.scm.ChangeLogParser;
14  
15  import java.io.File;
16  import java.io.FileReader;
17  import java.io.IOException;
18  import java.util.logging.Logger;
19  
20  import org.xml.sax.InputSource;
21  import org.xml.sax.SAXException;
22  import org.xml.sax.XMLReader;
23  import org.xml.sax.helpers.XMLReaderFactory;
24  
25  /**
26   * Parses the output of darcs log.
27   *
28   * @author Sven Strittmatter <ich@weltraumschaf.de>
29   */
30  public class DarcsChangeLogParser extends ChangeLogParser {
31  
32      /**
33       * Logger facility.
34       */
35      private static final Logger LOGGER = Logger.getLogger(DarcsChangeLogParser
36                                                            .class.getName());
37  
38      /**
39       * Parses the darcs changelog file.
40       *
41       * The darcs changelog file is in XML format (as given by the command
42       * darcs changes --xml-output --summary).
43       *
44       * @param build
45       * @param changelogFile
46       * @return
47       * @throws IOException
48       * @throws SAXException
49       */
50      @Override
51      public DarcsChangeSetList parse(AbstractBuild build, File changelogFile)
52      throws IOException, SAXException {
53          LOGGER.info("Parsing changelog file " + changelogFile.toString());
54  
55          XMLReader       xmlReader  = XMLReaderFactory.createXMLReader();
56          DarcsSaxHandler handler    = new DarcsSaxHandler();
57          FileReader      fileReader = new FileReader(changelogFile);
58  
59          xmlReader.setContentHandler(handler);
60          xmlReader.setErrorHandler(handler);
61          xmlReader.parse(new InputSource(fileReader));
62  
63          return new DarcsChangeSetList(build, handler.getChangeSets());
64      }
65  }
