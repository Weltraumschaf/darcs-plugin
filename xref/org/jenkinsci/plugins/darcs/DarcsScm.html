157         final TaskListener listener) throws IOException, InterruptedException {
158         final FilePath localPath = createLocalPath(build.getWorkspace());
159         final DarcsRevisionState local = getRevisionState(launcher, listener, localPath.getRemote());

166     @Override
167     protected PollingResult compareRemoteRevisionWith(final AbstractProject<?, ?> project, final Launcher launcher,
168         final FilePath workspace, final TaskListener listener, final SCMRevisionState baseline)
169         throws IOException, InterruptedException {
170         final PrintStream logger = listener.getLogger();
171         final DarcsRevisionState localRevisionState;
172 
173         if (baseline instanceof DarcsRevisionState) {
174             localRevisionState = (DarcsRevisionState) baseline;
175         } else if (project.getLastBuild() != null) {
176             localRevisionState = calcRevisionsFromBuild(project.getLastBuild(), launcher, listener);
177         } else {
178             localRevisionState = new DarcsRevisionState();
179         }
180 
181         if (null != project.getLastBuild()) {
182             logger.println("[poll] Last Build : #" + project.getLastBuild().getNumber());
183         } else {
184             // If we've never been built before, well, gotta build!
185             logger.println("[poll] No previous build, so forcing an initial build.");
186 
187             return PollingResult.BUILD_NOW;
188         }
189 
190         final Change change;
191         final DarcsRevisionState remoteRevisionState = getRevisionState(launcher, listener, source);
192 
193         logger.printf("[poll] Current remote revision is %s. Local revision is %s.%n",
194                 remoteRevisionState, localRevisionState);
195 
196         if (SCMRevisionState.NONE == localRevisionState) {
197             logger.println("[poll] Does not have a local revision state.");
198             change = Change.SIGNIFICANT;
199         } else if (localRevisionState.getClass() != DarcsRevisionState.class) {
200             // appears that other instances of None occur - its not a singleton.
201             // so do a (fugly) class check.
202             logger.println("[poll] local revision state is not of type darcs.");
203             change = Change.SIGNIFICANT;
204         } else if (!remoteRevisionState.equals(localRevisionState)) {
205             logger.println("[poll] Local revision state differs from remote.");
206 
207             if (remoteRevisionState.getChanges().size()
208                 < ((DarcsRevisionState) localRevisionState).getChanges().size()) {
209                 final FilePath ws = project.getLastBuild().getWorkspace();
210 
211                 logger.printf("[poll] Remote repo has less patches than local: remote(%s) vs. local(%s). Will wipe "
212                         + "workspace %s...%n",
213                         remoteRevisionState.getChanges().size(),
214                         ((DarcsRevisionState) localRevisionState).getChanges().size(),
215                         (null != ws) ? ws.getRemote() : "null");
216 
217                 if (null != ws) {
218                     ws.deleteRecursive();
219                 }
220             }
221 
222             change = Change.SIGNIFICANT;
223         } else {
224             change = Change.NONE;
225         }
226 
227         return new PollingResult(localRevisionState, remoteRevisionState, change);
228     }

230     /**
231      * Calculates the revision state of a repository (local or remote).
232      *
233      * @param launcher
234      * @param listener
235      * @param repo
236      * @return
237      * @throws InterruptedException
238      */
239     DarcsRevisionState getRevisionState(final Launcher launcher, final TaskListener listener, final String repo)
240         throws InterruptedException {
241         final DarcsCmd cmd;
242 
243         if (null == launcher) {
244             /* Create a launcher on master
245              * todo better grab a launcher on 'any slave'
246              */
247             cmd = new DarcsCmd(new LocalLauncher(listener), EnvVars.masterEnvVars, getDescriptor().getDarcsExe());
248         } else {
249             cmd = new DarcsCmd(launcher, EnvVars.masterEnvVars, getDescriptor().getDarcsExe());
250         }
251 
252         DarcsRevisionState rev = null;
253 
254         try {
255             final ByteArrayOutputStream changes = cmd.allChanges(repo);
256             changelogParser.parse(changes);
257             rev = new DarcsRevisionState(changelogParser.parse(changes));
258         } catch (Exception e) {
259             LOGGER.warning(String.format("Failed to get revision state for repository: %s", e));
260         }
261 
262         return rev;
263     }

265     /**
266      * Writes the change log of the last numPatches to the changeLog file.
267      *
268      * @param launcher
269      * @param numPatches
270      * @param workspace
271      * @param changeLog
272      * @throws InterruptedException
273      */
274     private void createChangeLog(final Launcher launcher, final int numPatches, final FilePath workspace,
275         final File changeLog, final BuildListener listener) throws InterruptedException {
276         if (0 == numPatches) {
277             LOGGER.info("Creating empty changelog.");
278             createEmptyChangeLog(changeLog, listener, "changelog");
279             return;
280         }
281 
282         final DarcsCmd cmd = new DarcsCmd(launcher, EnvVars.masterEnvVars,
283                 getDescriptor().getDarcsExe());
284         FileOutputStream fos = null;
285 
286         try {
287             fos = new FileOutputStream(changeLog);
288             final FilePath localPath = createLocalPath(workspace);
289             final ByteArrayOutputStream changes = cmd.lastSummarizedChanges(localPath.getRemote(), numPatches);
290             changes.writeTo(fos);
291         } catch (Exception e) {
292             final StringWriter w = new StringWriter();
293             e.printStackTrace(new PrintWriter(w));
294             LOGGER.warning(String.format("Failed to get log from repository: %s", w));
295         } finally {
296             if (null != fos) {
297                 try {
298                     fos.close();
299                 } catch (IOException ex) {
300                     throw new InterruptedException(String.format("Failed to close file output stream!%n%s", ex));
301                 }
302             }
303         }
304     }

306     @Override
307     public boolean checkout(final AbstractBuild<?, ?> build, final Launcher launcher, final FilePath workspace,
308         final BuildListener listener, final File changelogFile) throws IOException, InterruptedException {
309         final FilePath localPath = createLocalPath(workspace);
310         final boolean existsRepoinWorkspace = localPath.act(new FileCallable<Boolean>() {
311             private static final long serialVersionUID = 1L;
312 
313             public Boolean invoke(File ws, VirtualChannel channel) throws IOException {
314                 final File file = new File(ws, "_darcs");
315                 return file.exists();
316             }
317         });
318 
319         if (existsRepoinWorkspace && !isClean()) {
320             return pullRepo(build, launcher, workspace, listener, changelogFile);
321         } else {
322             return getRepo(build, launcher, workspace, listener, changelogFile);
323         }
324     }

327      * Counts the patches in a repository.
328      *
329      * @param build
330      * @param launcher
331      * @param workspace
332      * @param listener
333      * @return
334      * @throws InterruptedException
335      * @throws IOException
336      */
337     private int countPatches(final AbstractBuild<?, ?> build, final Launcher launcher, final FilePath workspace,
338         final BuildListener listener) {
339         try {
340             final DarcsCmd cmd = new DarcsCmd(launcher,
341                     build.getEnvironment(listener),
342                     getDescriptor().getDarcsExe());
343 
344             final FilePath localPath = createLocalPath(workspace);
345             return cmd.countChanges(localPath.getRemote());
346         } catch (Exception e) {
347             listener.error("Failed to count patches in workspace repo:%n", e.toString());
348             return 0;
349         }
350     }

352     /**
353      * Pulls all patches from a remote repository in the workspace repository.
354      *
355      * @param build
356      * @param launcher
357      * @param workspace
358      * @param listener
359      * @param changelogFile
360      * @return boolean
361      * @throws InterruptedException
362      * @throws IOException
363      */
364     private boolean pullRepo(final AbstractBuild<?, ?> build, final Launcher launcher, final FilePath workspace,
365         final BuildListener listener, final File changelogFile) throws InterruptedException, IOException {
366         LOGGER.info(String.format("Pulling repo from: %s", source));
367         final int preCnt = countPatches(build, launcher, workspace, listener);
368         LOGGER.info(String.format("Count of patches pre pulling is %d", preCnt));
369 
370         try {
371             final DarcsCmd cmd = new DarcsCmd(launcher,
372                     build.getEnvironment(listener),
373                     getDescriptor().getDarcsExe());
374             final FilePath localPath = createLocalPath(workspace);
375             cmd.pull(localPath.getRemote(), source);
376         } catch (Exception e) {
377             listener.error("Failed to pull: " + e.toString());
378             return false;
379         }
380 
381         final int postCnt = countPatches(build, launcher, workspace, listener);
382         LOGGER.info(String.format("Count of patches post pulling is %d", preCnt));
383         createChangeLog(launcher, postCnt - preCnt, workspace, changelogFile, listener);
384 
385         return true;
386     }

388     /**
389      * Gets a fresh copy of a remote repository.
390      *
391      * @param build
392      * @param launcher
393      * @param workspace
394      * @param listener
395      * @param changelogFile
396      * @return boolean
397      * @throws InterruptedException
398      */
399     private boolean getRepo(final AbstractBuild<?, ?> build, final Launcher launcher, final FilePath workspace,
400         final BuildListener listener, final File changeLog) throws InterruptedException {
401         LOGGER.info(String.format("Getting repo from: %s", source));
402 
403         try {
404             final FilePath localPath = createLocalPath(workspace);
405             localPath.deleteRecursive();
406         } catch (IOException e) {
407             e.printStackTrace(listener.error("Failed to clean the workspace"));
408             return false;
409         }
410 
411         try {
412             final DarcsCmd cmd = new DarcsCmd(launcher, build.getEnvironment(listener), getDescriptor().getDarcsExe());
413             final FilePath localPath = createLocalPath(workspace);
414             cmd.get(localPath.getRemote(), source);
415         } catch (Exception e) {
416             e.printStackTrace(listener.error("Failed to get repo from " + source));
417             return false;
418         }
419 
420         return createEmptyChangeLog(changeLog, listener, "changelog");
421     }

433     /**
434      * Creates a local path relative to the given base.
435      *
436      * If {@link #localDir} is not {@link null} and not empty a relative path to the given base is created,
437      * else the base pat itself is returned.     *
438      *
439      * @param base base of the local path
440      * @return local file path.
441      */
442     private FilePath createLocalPath(final FilePath base) {
443         if (null != localDir && localDir.length() > 0) {
444             return new FilePath(base, localDir);
445         }
446 
447         return base;
