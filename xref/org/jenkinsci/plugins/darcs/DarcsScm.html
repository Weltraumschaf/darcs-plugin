139     @Override
140     protected PollingResult compareRemoteRevisionWith(AbstractProject<?, ?> project, Launcher launcher, FilePath fp, TaskListener listener, SCMRevisionState localRevisionState) throws IOException, InterruptedException {
141 		final AbstractBuild lastBuild = project.getLastBuild();
142 
143         if (null != lastBuild) {
144             listener.getLogger()
145 					.println("[poll] Last Build : #" + lastBuild.getNumber());
146         } else {
147             // If we've never been built before, well, gotta build!
148             listener.getLogger()
149 					.println("[poll] No previous build, so forcing an initial build.");
150 			
151             return PollingResult.BUILD_NOW;
152         }
153 		
154         final Change change;
155         final DarcsRevisionState remoteRevisionState = getRevisionState(launcher, listener, source);
156 
157         listener.getLogger().printf("[poll] Current remote revision is %s. Baseline is %s.\n", 
158 									remoteRevisionState, 
159 									localRevisionState);
160         
161         if ((SCMRevisionState.NONE == localRevisionState)
162             // appears that other instances of None occur - its not a singleton.
163             // so do a (fugly) class check.
164             || (localRevisionState.getClass() != DarcsRevisionState.class)
165             || (!remoteRevisionState.equals(localRevisionState))) {
166             change = Change.SIGNIFICANT;
167         } else {
168             change = Change.NONE;
169         }
170 
171         return new PollingResult(localRevisionState, remoteRevisionState, change);
172     }

174     /**
175      * Calculates the revision state of a repository (local or remote).
176      *
177      * @param launcher
178      * @param listener
179      * @param repo
180      * @return
181      * @throws InterruptedException
182      */
183     protected DarcsRevisionState getRevisionState(Launcher launcher, TaskListener listener, String repo) throws InterruptedException {
184         DarcsRevisionState rev = null;
185 
186         if (null == launcher) {
187             /* Create a launcher on master
188              * todo better grab a launcher on 'any slave'
189              */
190             launcher = new LocalLauncher(listener);
191         }
192 
193         DarcsCmd cmd = new DarcsCmd(launcher, EnvVars.masterEnvVars,
194                                     getDescriptor().getDarcsExe());
195 
196         try {
197             byte[]			changes = cmd.allChanges(repo).toByteArray();
198             XMLReader       xr      = XMLReaderFactory.createXMLReader();
199             DarcsSaxHandler handler = new DarcsSaxHandler();
200 
201             xr.setContentHandler(handler);
202             xr.setErrorHandler(handler);
203             xr.parse(new InputSource(new ByteArrayInputStream(changes)));
204 
205             rev = new DarcsRevisionState(new DarcsChangeSetList(null, handler.getChangeSets()));
206         } catch (Exception e) {
207             StringWriter w = new StringWriter();
208             e.printStackTrace(new PrintWriter(w));
209             LOGGER.log(Level.WARNING, "Failed to get revision state for repository: ", e);
210         }
211 
212         return rev;
213     }

215     /**
216      * Writes the changelog of the last numPatches to the changeLog file.
217      *
218      * @param launcher
219      * @param numPatches
220      * @param workspace
221      * @param changeLog
222      * @throws InterruptedException
223      */
224     private void createChangeLog(Launcher launcher, int numPatches, FilePath workspace, File changeLog, BuildListener listener) throws InterruptedException {
225         if (0 == numPatches) {
226             LOGGER.info("Creating empty changelog.");
227             createEmptyChangeLog(changeLog, listener, "changelog");
228             return;
229         }
230 
231         DarcsCmd cmd = new DarcsCmd(launcher, EnvVars.masterEnvVars,
232                                     getDescriptor().getDarcsExe());
233 
234         try {
235             FileOutputStream fos          = new FileOutputStream(changeLog);
236             ByteArrayOutputStream changes = cmd.lastSummarizedChanges(workspace.getRemote(), numPatches);
237 
238             changes.writeTo(fos);
239             fos.close();
240         } catch (Exception e) {
241             StringWriter w = new StringWriter();
242             e.printStackTrace(new PrintWriter(w));
243             LOGGER.log(Level.WARNING, "Failed to get log from repository: ", e);
244         }
245     }

247     @Override
248     public boolean checkout(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changelogFile) throws IOException, InterruptedException {
249         boolean existsRepoinWorkspace = workspace.act(new FileCallable<Boolean>() {
250 
251             private static final long serialVersionUID = 1L;
252 
253             public Boolean invoke(File ws, VirtualChannel channel) throws IOException {
254                 File file = new File(ws, "_darcs");
255                 return file.exists();
256             }
257         });
258 
259         if (existsRepoinWorkspace && !isClean()) {
260             return pullRepo(build, launcher, workspace, listener, changelogFile);
261         } else {
262             return getRepo(build, launcher, workspace, listener, changelogFile);
263         }
264     }

290     /**
291      * Pulls all patches from a remote repo in the workspace repo.
292      *
293      * @param build
294      * @param launcher
295      * @param workspace
296      * @param listener
297      * @param changelogFile
298      * @return boolean
299      * @throws InterruptedException
300      * @throws IOException
301      */
302     private boolean pullRepo(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changelogFile) throws InterruptedException, IOException {
303         LOGGER.log(Level.INFO, "Pulling repo from: {0}", source);
304         int preCnt = 0, postCnt = 0;
305         preCnt = countPatches(build, launcher, workspace, listener);
306         LOGGER.log(Level.INFO, "Count of patches pre pulling is {0}", preCnt);
307 
308         try {
309             DarcsCmd cmd = new DarcsCmd(launcher,
310                                         build.getEnvironment(listener),
311                                         getDescriptor().getDarcsExe());
312             cmd.pull(workspace.getRemote(), source);
313         } catch (Exception e) {
314             listener.error("Failed to pull: " + e.toString());
315 
316             return false;
317         }
318 
319         postCnt = countPatches(build, launcher, workspace, listener);
320         LOGGER.log(Level.INFO, "Count of patches post pulling is {0}", preCnt);
321         createChangeLog(launcher, postCnt - preCnt, workspace, changelogFile, listener);
322 
323         return true;
324     }

326     /**
327      * Gets a fresh copy of a remote repo.
328      *
329      * @param build
330      * @param launcher
331      * @param workspace
332      * @param listener
333      * @param changelogFile
334      * @return boolean
335      * @throws InterruptedException
336      */
337     private boolean getRepo(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changeLog) throws InterruptedException {
338         LOGGER.log(Level.INFO, "Getting repo from: {0}", source);
339 
340         try {
341             workspace.deleteRecursive();
342         } catch (IOException e) {
343             e.printStackTrace(listener.error("Failed to clean the workspace"));
344             return false;
345         }
346 
347         try {
348             DarcsCmd cmd = new DarcsCmd(launcher,
349                                         build.getEnvironment(listener),
350                                         getDescriptor().getDarcsExe());
351             cmd.get(workspace.getRemote(), source);
352         } catch (Exception e) {
353             e.printStackTrace(listener.error("Failed to get repo from " + source));
354 
355             return false;
356         }
357 
358         return createEmptyChangeLog(changeLog, listener, "changelog");
359     }

414         public FormValidation doDarcsExeCheck(@QueryParameter final String value) throws IOException, ServletException {
415             return FormValidation.validateExecutable(value, new FormValidation.FileValidator() {
416 
417                 @Override
418                 public FormValidation validate(File exe) {
419                     try {
420                         ByteBuffer  baos     = new ByteBuffer();
421                         Launcher    launcher = Hudson.getInstance()
422                                                      .createLauncher(TaskListener.NULL);
423                         ProcStarter proc     = launcher.launch()
424                                                        .cmds(exe, "--version")
425                                                        .stdout(baos);
426 
427                         if (proc.join() == 0) {
428                             return FormValidation.ok();
429                         } else {
430                             return FormValidation.warning("Could not locate the executable in path");
431                         }
432                     } catch (IOException e) {
433                         // failed
434                         LOGGER.log(Level.WARNING, e.toString());
435                     } catch (InterruptedException e) {
436                         // failed
437                         LOGGER.log(Level.WARNING, e.toString());
438                     }
439 
440                     return FormValidation.error("Unable to check darcs version");
441                 }
442             });
