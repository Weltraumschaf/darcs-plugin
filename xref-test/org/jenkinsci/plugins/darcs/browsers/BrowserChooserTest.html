83      JSONObject readJson() throws IOException {
84          final InputStream stream = this.getClass().getResourceAsStream("scm.json");
85          final String scmString;
86  
87          try {
88              scmString = IOUtils.toString(stream);
89          } finally {
90              stream.close();
91          }
92  
93          final JSONObject json = (JSONObject) JSONSerializer.toJSON(scmString);
94  
95          return json;
96      }
