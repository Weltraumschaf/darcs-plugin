33              super(new URL(URL), REPO);
34          }
35  
36          public DarcsQueryBuilder exposedCreateDefaultQuery() {
37              return createDefaultQuery();
38          }

40          public DarcsQueryBuilder exposedCreateDefaultQuery(String action) {

42          }
43      }

49      public void testCreateDefaultQuery() {
50          try {
51              ExposingDarcsWeb sut = new ExposingDarcsWeb();
52              assertEquals("Query",
53                           "?r=" + sut.repo,
54                           sut.exposedCreateDefaultQuery().toString());
55          } catch (MalformedURLException e) {
56              fail("Can not create SUT!");
57          }
58      }
